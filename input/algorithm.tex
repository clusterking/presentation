\section{Algorithm}

\begin{frame}{Algorithm}
	\stress{Quantify \enquote{similar}} distributions:\\
	$\lra$ similarity test, e.g. $\chi^2$ test or Kolmogorov test
	
	
	
	\bigskip
	\stress{Build up groups} (\emph{clusters}) of similar distributions:\\
	Need a clustering algorithm, e.g. hierarchical clustering
	
	\bigskip
	Need to know \stress{how fine} (or coarse) our clustering should be:\\
	$\lra$ Add experimental error expectation\\
	$\lra$ Keep as many clusters as we can distinguish
	
	

\end{frame}
